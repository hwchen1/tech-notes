            params="8B",

        )

# End-to-End Streaming Pipeline
def streaming_inference(model, image, audio_stream):
    # 1. Encode image (one-shot)
    vis_tokens = model.vision_encoder(image)

    # 2. Encode audio (streaming chunks)
    for chunk in audio_stream:
        audio_tokens = model.audio_encoder(chunk)

        # 3. LLM processes combined input
        output = model.llm(
            vis_tokens, audio_tokens
        )

        # 4. Generate text + speech simultaneously
        text_out = output.text_logits
        speech_out = model.tts_decoder(
            output.speech_codes
        )
        yield text_out, speech_out

# MiniCPM-o: Multimodal Encoding Pipeline
# Image + Audio + Text → Unified Token Space

# === Vision Encoding (SigLIP + Adaptive Slice) ===
class VisionEncoder(nn.Module):
    def __init__(self):
        self.siglip = SigLIP_ViT(
            patch_size=14,
            hidden_dim=1152,
        )
        self.projector = nn.Linear(
            1152, 4096  # → LLM dim
        )

    def forward(self, image):
        # Adaptive resolution slicing
        # (like LLaVA-NeXT / InternVL)
        slices = adaptive_slice(
            image,
            max_res=1344,  # max 1344x1344
            grid_options=[
                (1,1), (1,2), (2,1),
                (2,2), (1,3), (3,1),
            ]
        )
        # Each slice → 448x448 → SigLIP

        all_tokens = []
        for s in slices:
            feat = self.siglip(s)
            # [num_patches, 1152]
            tokens = self.projector(feat)
            all_tokens.append(tokens)

        return torch.cat(all_tokens)
        # → visual_tokens in LLM space

# === Audio Encoding (Whisper + Streaming) ===
class AudioEncoder(nn.Module):
    def __init__(self):
        self.whisper = WhisperEncoder(
            n_mels=128,
            d_model=1024,
        )
        self.projector = nn.Linear(
            1024, 4096  # → LLM dim
        )

    def forward(self, audio_chunk):
        # Streaming: process in chunks
        # chunk_size ≈ 0.5~2s of audio
        mel = log_mel_spectrogram(
            audio_chunk, sr=16000
        )
        feat = self.whisper(mel)
        # [T, 1024] — T depends on duration
        tokens = self.projector(feat)
        return tokens
        # → audio_tokens in LLM space

Adaptive Resolution (Vision)

Dynamic slice strategy (similar to LLaVA-NeXT): input image is split into optimal grid (1x1, 1x2, 2x2, etc.) based on aspect ratio. Each slice is resized to 448x448 and processed by SigLIP independently.

$N_{\text{vis}} = N_{\text{slices}} \times \left(\frac{448}{14}\right)^2 = N_s \times 1024$

Streaming Audio

Chunked processing: Audio is split into overlapping chunks (~0.5-2s), enabling real-time turn-taking without waiting for utterance completion.

This enables sub-second latency for voice conversations.

Interleaved Format

Special tokens [IMG_START], [IMG_END], [AUD_START], [AUD_END] delineate modality boundaries in the unified sequence.

3. 端到端语音输出 + 训练

# MiniCPM-o: End-to-End Speech Generation
# LLM directly generates speech codes (no TTS post-processing)

# === Speech Code Generation ===
class SpeechOutputHead(nn.Module):
    def __init__(self):
        # Speech codebook (learned discrete codes)
        self.speech_head = nn.Linear(
            4096, 4096  # codebook size
        )
        # The LLM learns to output speech codes
        # alongside text tokens!

    def forward(self, hidden_states):
        # hidden: [B, T, 4096] from LLM
        speech_logits = self.speech_head(hidden_states)
        # [B, T, 4096] — codebook distribution
        return speech_logits

# === Speech Decoder (Codes → Waveform) ===
class SpeechDecoder(nn.Module):
    def __init__(self):
        self.codebook = nn.Embedding(
            4096, 512
        )
        self.vocoder = ConvDecoder(
            # codes → mel → waveform
            upsample_rates=[8, 5, 4, 2],
            # total upsample: 320x
        )

    def decode(self, speech_codes):
        # speech_codes: [B, T_speech]
        embeddings = self.codebook(speech_codes)
        waveform = self.vocoder(embeddings)
        # [B, T_audio] — raw audio output
        return waveform

End-to-End Speech Innovation

Key difference from traditional systems: No separate TTS pipeline. The LLM itself generates discrete speech codes that are decoded into audio waveform.

The model learns: WHEN to speak, WHAT to say, and HOW to say it (tone, emotion, pacing) — all from a single unified LLM.

Training Loss

$\mathcal{L}_{\text{total}} = \mathcal{L}_{\text{text}}^{\text{NTP}} + \lambda \cdot \mathcal{L}_{\text{speech}}^{\text{NTP}}$

Both text and speech generation are trained with next-token prediction. The speech codes are treated as a separate "language" that the model learns jointly.

Training Stages

1. Vision Pretrain: Align visual encoder with LLM (projector only)

2. Audio Pretrain: Align audio encoder + speech output (projector + speech head)

3. Omni Joint: Full model training on mixed multimodal data

4. RLHF: Align for naturalness, safety, and user preference
